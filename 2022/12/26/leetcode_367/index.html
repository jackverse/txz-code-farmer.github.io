/**
* 二分法
* @param num
* @return
*/
public boolean isPerfectSquare(int num) {
int left = 0, right = num;
while (left <= right) {
int middle = left + (right - left) / 2;
if ((long) middle * middle > num) {
right = middle - 1;
} else if ((long) middle * middle < num) {
left = middle + 1;
} else {
return true;
}
}
return false;
}
